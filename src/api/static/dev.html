Integration Example (JavaScript)

const query = `{
  getHistory(tokenAddress: "...", interval: "1m") {
    time
    close
    volume
  }
}`;

  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ query })
}).then(res => res.json());
